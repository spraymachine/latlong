# LatLong V1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) and superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a mobile-first web app where authenticated sailors create public A-to-B voyages and publish exact-coordinate compressed photo posts onto a shared public ocean map.

**Architecture:** Use a root workspace with a single Next.js app in `apps/web`, Supabase for auth/database/storage, and a map-first public UI backed by RLS-protected voyage/post tables. Keep uploads online-only, compress every image in the browser before upload, and render voyage routes as a simple ordered line from start point through post points to end point.

**Tech Stack:** `pnpm` workspace, Next.js App Router, TypeScript, Tailwind CSS, Supabase Auth/Postgres/Storage, MapLibre GL JS, Zod, React Hook Form, browser-image-compression, Vitest, React Testing Library, Playwright

```sql
-- supabase/migrations/20260414_init_latlong.sql
create table public.profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  display_name text not null,
  created_at timestamptz not null default now()
);

create table public.voyages (
  id uuid primary key default gen_random_uuid(),
  user_id uuid not null references public.profiles(id) on delete cascade,
  title text not null,
  description text,
  start_name text not null,
  start_latitude double precision not null,
  start_longitude double precision not null,
  end_name text not null,
  end_latitude double precision not null,
  end_longitude double precision not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table public.posts (
  id uuid primary key default gen_random_uuid(),
  voyage_id uuid not null references public.voyages(id) on delete cascade,
  user_id uuid not null references public.profiles(id) on delete cascade,
  image_path text not null,
  caption text,
  latitude double precision not null,
  longitude double precision not null,
  posted_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

alter table public.profiles enable row level security;
alter table public.voyages enable row level security;
alter table public.posts enable row level security;

create policy "profiles are public read" on public.profiles for select using (true);
create policy "voyages are public read" on public.voyages for select using (true);
create policy "posts are public read" on public.posts for select using (true);
create policy "owners manage voyages" on public.voyages for all using (auth.uid() = user_id) with check (auth.uid() = user_id);
create policy "owners manage posts" on public.posts for all using (auth.uid() = user_id) with check (auth.uid() = user_id);

insert into storage.buckets (id, name, public) values ('voyage-photos', 'voyage-photos', true);
create policy "public read voyage photos" on storage.objects for select using (bucket_id = 'voyage-photos');
create policy "owners upload voyage photos" on storage.objects for insert with check (bucket_id = 'voyage-photos' and auth.role() = 'authenticated');
```

```ts
// apps/web/src/app/(auth)/actions.ts
'use server'

import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'

export async function signUpAction(formData: FormData) {
  const supabase = await createClient()
  const email = String(formData.get('email'))
  const password = String(formData.get('password'))
  const displayName = String(formData.get('displayName'))

  const { data, error } = await supabase.auth.signUp({
    email,
    password,
    options: {
      data: { display_name: displayName },
    },
  })

  if (error) throw error

  await supabase.from('profiles').upsert({
    id: data.user!.id,
    display_name: displayName,
  })

  redirect('/dashboard')
}

export async function signInAction(formData: FormData) {
  const supabase = await createClient()
  const email = String(formData.get('email'))
  const password = String(formData.get('password'))

  const { error } = await supabase.auth.signInWithPassword({ email, password })
  if (error) throw error

  redirect('/dashboard')
}
```

```ts
import { createClient } from '@/lib/supabase/server'

export async function getDashboardData() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()

  if (!user) {
    return { user: null, voyages: [] }
  }

  const { data: voyages } = await supabase
    .from('voyages')
    .select('id, title, start_name, end_name, created_at')
    .eq('user_id', user.id)
    .order('created_at', { ascending: false })

  return { user, voyages: voyages ?? [] }
}
```

```ts
// apps/web/src/app/(dashboard)/voyages/actions.ts
'use server'

import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { voyageSchema } from '@/lib/validation/voyage'

export async function createVoyageAction(formData: FormData) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/sign-in')

  const parsed = voyageSchema.parse({
    title: formData.get('title'),
    description: formData.get('description'),
    startName: formData.get('startName'),
    startLatitude: formData.get('startLatitude'),
    startLongitude: formData.get('startLongitude'),
    endName: formData.get('endName'),
    endLatitude: formData.get('endLatitude'),
    endLongitude: formData.get('endLongitude'),
  })

  const { data, error } = await supabase
    .from('voyages')
    .insert({
      user_id: user.id,
      title: parsed.title,
      description: parsed.description,
      start_name: parsed.startName,
      start_latitude: parsed.startLatitude,
      start_longitude: parsed.startLongitude,
      end_name: parsed.endName,
      end_latitude: parsed.endLatitude,
      end_longitude: parsed.endLongitude,
    })
    .select('id')
    .single()

  if (error) throw error

  redirect(`/voyages/${data.id}`)
}
```

```ts
// apps/web/src/lib/geo/route.ts
type Coordinate = { latitude: number; longitude: number }
type RoutePoint = Coordinate & { postedAt: string }

export function buildVoyageLine(start: Coordinate, posts: RoutePoint[], end: Coordinate) {
  const orderedPosts = [...posts].sort((a, b) => a.postedAt.localeCompare(b.postedAt))

  return {
    type: 'LineString' as const,
    coordinates: [
      [start.longitude, start.latitude],
      ...orderedPosts.map((post) => [post.longitude, post.latitude]),
      [end.longitude, end.latitude],
    ],
  }
}
```

```tsx
// apps/web/src/components/posts/post-form.tsx
'use client'

import { useEffect, useState } from 'react'
import { compressImage } from '@/lib/media/compress-image'
import { createClient } from '@/lib/supabase/browser'
import { createPostRecordAction } from '@/app/(dashboard)/voyages/[voyageId]/posts/actions'

export default function PostForm({ voyageId }: { voyageId: string }) {
  const [latitude, setLatitude] = useState('')
  const [longitude, setLongitude] = useState('')
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    navigator.geolocation.getCurrentPosition(
      (position) => {
        setLatitude(String(position.coords.latitude))
        setLongitude(String(position.coords.longitude))
      },
      () => {},
      { enableHighAccuracy: true, maximumAge: 60000 },
    )
  }, [])

  async function handleSubmit(event: React.FormEvent<HTMLFormElement>) {
    event.preventDefault()
    setError(null)

    const formData = new FormData(event.currentTarget)
    const file = formData.get('photo')
    if (!(file instanceof File)) {
      setError('Choose a photo before publishing.')
      return
    }

    try {
      const compressed = await compressImage(file)
      const supabase = createClient()
      const filePath = `${voyageId}/${crypto.randomUUID()}.webp`

      const { error: uploadError } = await supabase.storage
        .from('voyage-photos')
        .upload(filePath, compressed, { contentType: 'image/webp', upsert: false })

      if (uploadError) throw uploadError

      await createPostRecordAction({
        voyageId,
        caption: String(formData.get('caption') ?? ''),
        latitude: Number(latitude),
        longitude: Number(longitude),
        imagePath: filePath,
      })
    } catch {
      setError('Upload failed. Keep your coordinates, choose the photo again, and retry.')
    }
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <label>
        Photo
        <input name="photo" type="file" accept="image/*" required />
      </label>
      <label>
        Latitude
        <input name="latitude" value={latitude} onChange={(event) => setLatitude(event.target.value)} required />
      </label>
      <label>
        Longitude
        <input name="longitude" value={longitude} onChange={(event) => setLongitude(event.target.value)} required />
      </label>
      <label>
        Caption
        <textarea name="caption" maxLength={280} />
      </label>
      {error ? <p className="rounded-xl bg-red-950/60 p-3 text-sm text-red-200">{error}</p> : null}
      <button type="submit">Publish point</button>
    </form>
  )
}
```

```ts
// apps/web/src/app/(dashboard)/voyages/[voyageId]/posts/actions.ts
'use server'

import { revalidatePath } from 'next/cache'
import { createClient } from '@/lib/supabase/server'
import { postSchema } from '@/lib/validation/post'

export async function createPostRecordAction(input: {
  voyageId: string
  caption: string
  latitude: number
  longitude: number
  imagePath: string
}) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) throw new Error('Authentication required')

  const parsed = postSchema.parse({
    voyageId: input.voyageId,
    caption: input.caption,
    latitude: input.latitude,
    longitude: input.longitude,
    fileName: input.imagePath,
    contentType: 'image/webp',
  })

  const { data: voyage, error: voyageError } = await supabase
    .from('voyages')
    .select('id, user_id')
    .eq('id', parsed.voyageId)
    .eq('user_id', user.id)
    .single()

  if (voyageError || !voyage) throw new Error('Voyage not found')

  const { error } = await supabase.from('posts').insert({
    voyage_id: parsed.voyageId,
    user_id: user.id,
    caption: parsed.caption,
    latitude: parsed.latitude,
    longitude: parsed.longitude,
    image_path: input.imagePath,
  })

  if (error) throw error

  revalidatePath('/')
  revalidatePath(`/voyages/${parsed.voyageId}`)
}
```
